\section{Theorem 2: Sufficiency of Dual-Timescale Adaptation}
\label{sec:theorem2}

\begin{remark}[Scope restriction]
\label{rem:thm2-scope}
The following results are stated and proved for the
\emph{linear-Gaussian model}: $y_t = \theta^{*\top}_t x_t + \epsilon_t$,
$x_t \sim \mathcal{N}(0, \Sigma_t)$, $\epsilon_t \sim \mathcal{N}(0,
\sigma_\epsilon^2)$, with $\mu I \preceq \Sigma_t \preceq LI$.
In this setting, $L_t(\theta) = \frac{1}{2}\theta^\top\Sigma_t\theta
- \theta^\top \Sigma_t\theta^*_t + \text{const}$ is $\mu$-strongly convex
and $L$-smooth with constant Hessian $\Sigma_t$, and the OLS estimator
has a closed form. Extension to general strongly convex losses would
require replacing OLS with online Newton steps and is left to future work.
\end{remark}

%% ======================================================================
\subsection{Dual-Timescale Algorithm}
%% ======================================================================

\begin{definition}[Dual-timescale learner]
\label{def:dual-timescale}
Given a subspace estimate $\hat{U}_t \in \R^{d \times r}$ (orthonormal
columns) and a consolidation period $B \in \N$, the dual-timescale
learner maintains:

\begin{enumerate}[nosep]
  \item \textbf{Fast track (coefficient estimation):} A sliding-window
    OLS estimator with window size $w$:
    \[
      \hat{a}_t = \argmin_{a \in \R^r}
      \sum_{s=\max(1,t-w+1)}^{t} \ell_s(\hat{U}_t\, a).
    \]
    For the linear model $\ell_s(U a) = \frac{1}{2}(a^\top U^\top x_s - y_s)^2$,
    this has closed form:
    \[
      \hat{a}_t = \Bigl(\sum_{s \in W_t} z_s z_s^\top\Bigr)^{-1}
      \sum_{s \in W_t} z_s y_s,
      \qquad z_s = \hat{U}_t^\top x_s,
    \]
    where $W_t = \{t-w+1, \ldots, t\}$.

  \item \textbf{Slow track (subspace update):} Every $B$ steps, update
    the projection matrix $\hat{P}_t = \hat{U}_t\hat{U}_t^\top$ via
    projected gradient descent on the set of rank-$r$ projectors
    $\mathcal{P}_r$:
    \[
      \hat{P}_{(m+1)B} = \mathrm{Proj}_{\mathcal{P}_r}\!\left(
        \hat{P}_{mB} - \eta_s \nabla_P f_{mB}(\hat{P}_{mB})
      \right),
    \]
    where $f_{mB}(P)$ is the empirical loss over a recent batch
    (see \Cref{def:subspace-loss} below), the projection
    $\mathrm{Proj}_{\mathcal{P}_r}$ retains the top-$r$ eigenvectors,
    and $\eta_s$ is the slow step size. Between consolidations,
    $\hat{P}_t = \hat{P}_{mB}$ for $t \in [mB, (m+1)B)$.
    The basis $\hat{U}_t$ is extracted from $\hat{P}_t$ via eigendecomposition.
\end{enumerate}
The full parameter estimate is $\hat{\theta}_t = \hat{U}_t\, \hat{a}_t$.
\end{definition}

\subsection{Version A: General Variation Bound}
%% ======================================================================

\begin{theorem}[Sufficiency --- general variation, honest bound]
\label{thm:sufficiency-A}
Under \Cref{ass:strong-convexity,ass:smoothness,ass:noise,%
ass:decomposition,ass:bounded-variation}, the dual-timescale learner
(\Cref{def:dual-timescale}) with window size $w$ and consolidation
period $B$ achieves:
\[
  \E[\reg_T] \leq
  \underbrace{\frac{L\sigma^2 r w}{2\mu}}_{\text{noise floor}} \cdot \frac{T}{w}
  + \underbrace{\frac{L}{2}\, V_f(T) \cdot w}_{\text{fast tracking lag}}
  + \underbrace{\frac{L}{\mu}\sqrt{T\, V_s(T)}}_{\text{slow tracking}}
  + \underbrace{L\rho_{\max}^2 T}_{\text{residual}}.
\]
With explicit dependence on all parameters:
\begin{align}
  \E[\reg_T] &\leq
    \frac{L\sigma^2 r\, T}{2\mu}
    + \frac{Lw\, V_f(T)}{2}
    + \frac{L}{\mu}\sqrt{T\, V_s(T)}
    + L\rho_{\max}^2 T.
  \label{eq:thm2-versionA}
\end{align}
Under the standard drift regime $V_f(T) = \Theta(\sqrt{T})$,
$V_s(T) = O(T^{1/2-\epsilon})$, with $w = O(1)$:
\[
  \E[\reg_T] = O(T) + O(\sqrt{T}) + O(T^{1/2}) + O(T)
  = O(T).
\]
This does \textbf{not} give sub-linear regret in general. For
sub-linear bounds, we need the jump-process structure.
\end{theorem}

\begin{proof}
Decompose the instantaneous regret:
\begin{align*}
  L_t(\hat{\theta}_t) - L_t(\theta^*_t)
  &\leq \frac{L}{2}\norm{\hat{\theta}_t - \theta^*_t}^2 \\
  &= \frac{L}{2}\norm{\hat{U}_t \hat{a}_t - U_t a_t - \rho_t}^2 \\
  &\leq L\bigl(\norm{\hat{U}_t \hat{a}_t - U_t a_t}^2 + \rho_{\max}^2\bigr) \\
  &\leq L\bigl(\norm{(\hat{U}_t - U_t)\hat{a}_t}^2
    + \norm{U_t(\hat{a}_t - a_t)}^2
    + 2\norm{(\hat{U}_t-U_t)\hat{a}_t}\norm{U_t(\hat{a}_t-a_t)}
    + \rho_{\max}^2\bigr).
\end{align*}

\textbf{Fast-track error (derived for linear-Gaussian model).}
Within window $W_t$, if $a_s = a_t$ for all $s \in W_t$ (no jump within
the window) and $U_s = U_t$ for all $s \in W_t$ (subspace constant within
window, valid when $w \cdot \rho_s \ll 1$), the OLS estimator
$\hat{a}_t = (\hat\Sigma_z)^{-1}\hat{b}_z$ where
$\hat\Sigma_z = \frac{1}{w}\sum_{s \in W_t} z_s z_s^\top$,
$z_s = \hat{U}_t^\top x_s$, satisfies:

\emph{Variance term}: $\E[\norm{\hat{a}_t - a^*(\hat{U}_t)}^2]
\leq \sigma^2 \tr(\Sigma_z^{-1})/w \leq \sigma^2 r/(\mu w)$
(by $\lambda_{\min}(\Sigma_z) = \lambda_{\min}(\hat{U}_t^\top\Sigma_t\hat{U}_t) \geq \mu$ when $w \geq 2r/\mu$ ensures
$\hat\Sigma_z \succeq \frac{\mu}{2}I_r$ w.h.p.).

\emph{Bias term}: The misspecification error from projecting onto
$\hat{U}_t$ instead of $U_t$ is:
\begin{align*}
  \norm{a^*(\hat{U}_t) - a_t}^2
  &= \norm{(\hat{U}_t^\top\Sigma_t\hat{U}_t)^{-1}
    \hat{U}_t^\top\Sigma_t U_t a_t - a_t}^2 \\
  &\leq \frac{L^2 A_{\max}^2}{\mu^2}\,
    \dG^2(\col(\hat{U}_t), \col(U_t)),
\end{align*}
using $\norm{U_t^\top(\hat{U}_t\hat{U}_t^\top - U_tU_t^\top)} \leq
\norm{\hat{U}_t\hat{U}_t^\top - U_tU_t^\top}_F = \sqrt{2}\dG$
(by \Cref{lem:grassmann-props}(3)) and the bounded coefficient
$\norm{a_t} \leq A_{\max}$.

Combined: $\E[\norm{\hat{a}_t - a_t}^2 \mid \hat{U}_t]
\leq \frac{\sigma^2 r}{\mu w}
+ \frac{2L^2 A_{\max}^2}{\mu^2}\dG^2(\col(\hat{U}_t), \col(U_t))$.

\emph{Bounded $\hat{a}_t$}: Since $\hat{a}_t = a^*(\hat{U}_t) +
\text{noise}$, and $\norm{a^*(\hat{U}_t)} \leq (L/\mu)A_{\max}$,
we have $\E[\norm{\hat{a}_t}^2] \leq 2\kappa^2 A_{\max}^2
+ 2\sigma^2 r/(\mu w)$, which is bounded.

If a jump occurs within $W_t$, the window contains inconsistent data.
The worst-case additional error is bounded by $\Delta_{\max}^2$
where $\Delta_{\max} = 2A_{\max}$.

\begin{definition}[Subspace loss]
\label{def:subspace-loss}
For a rank-$r$ projection matrix $P \in \mathcal{P}_r$ and data
$(x, y)$, define the subspace loss as the best achievable loss
in the column space of $P$:
\[
  f_t(P) \coloneqq \min_{a \in \R^r} L_t(U_P a),
\]
where $U_P \in \R^{d \times r}$ is any orthonormal basis of $\col(P)$.
For the linear-Gaussian model:
$f_t(P) = -\frac{1}{2}b_t^\top P(P\Sigma_t P)^+ P b_t + \mathrm{const}$,
where $b_t = \Sigma_t\theta^*_t$ and $(P\Sigma_t P)^+$ is the
pseudoinverse on $\col(P)$.
\end{definition}

\textbf{Slow-track error (via projection matrices).}
We work with projection matrices $P_t = U_t U_t^\top \in \R^{d \times d}$.

\emph{Convexity note}: The set $\mathcal{P}_r$ of rank-$r$ orthogonal
projectors is \textbf{not} convex (a convex combination of two rank-$r$
projectors may have rank $> r$). We therefore use the standard
\emph{Riemannian online gradient descent} framework on the Grassmannian
manifold $\Gr(d,r)$ \textup{(cf.\ Bonnabel, \emph{COLT} 2013;
Zhang \& Sra, \emph{NeurIPS} 2016)}. For geodesically smooth
losses, the dynamic regret bound is:

Online gradient descent on projection matrices with step size
$\eta_s$ and projection $\mathrm{Proj}_{\mathcal{P}_r}$ (eigenvalue
thresholding) achieves (by standard OGD on convex losses over a convex set
of diameter $D = \sqrt{2r}$):
\[
  \sum_{m=0}^{M-1} \bigl(\hat{L}_{mB}(\hat{P}_{mB})
  - \hat{L}_{mB}(P_{mB})\bigr)
  \leq \frac{D^2}{2\eta_s} + \frac{\eta_s}{2}\sum_{m=0}^{M-1}
    \norm{\hat{g}_m}^2 + D\sum_{m=0}^{M-1}\norm{P_{mB}-P_{(m-1)B}}_F,
\]
where $M = T/B$ and the last term accounts for comparator drift
(dynamic regret of OGD). Since $\norm{P_{mB}-P_{(m-1)B}}_F
= \sqrt{2}\dG(\col(U_{mB}), \col(U_{(m-1)B}))$ by
\Cref{lem:grassmann-props}(3), summing gives
$\sqrt{2}\,V_s(T)$.

With $\E[\norm{\hat{g}_m}^2] \leq G_s^2 = L^2 A_{\max}^2 + \sigma_s^2$
and optimizing $\eta_s = D\sqrt{V_s(T)}/(G_s\sqrt{M})$:
\[
  \sum_{m=0}^{M-1} \E[\dG^2(\hat{U}_{mB}, U_{mB})]
  \leq \frac{2}{\mu}\left(
    D\, G_s \sqrt{M\, V_s(T)} + D\sqrt{2}\,V_s(T)
  \right)
  = O\!\left(\frac{\kappa A_{\max}}{\mu}
    \sqrt{\frac{T\, V_s(T)}{B}}\right).
\]

\textbf{Combining (cross-term handling).}
By $(a+b)^2 \leq 2a^2 + 2b^2$:
\[
  \norm{\hat{U}_t\hat{a}_t - U_t a_t}^2
  \leq 2\norm{(\hat{U}_t-U_t)\hat{a}_t}^2
  + 2\norm{U_t(\hat{a}_t - a_t)}^2.
\]
The first term: using the bounded estimate
$\norm{\hat{a}_t} \leq \bar{A} \coloneqq 2\kappa A_{\max}$
and the Procrustes-aligned basis (let $R^*_t$ be the Procrustes
rotation aligning $\hat{U}_t$ to $U_t$, so
$\hat{U}_t = U_t R^*_t + E_t$ with $\norm{E_t}_F \leq \sqrt{2}\dG$
by \Cref{lem:grassmann-props}(4)):
\begin{align*}
  \norm{(\hat{U}_t - U_tR^*_t)\hat{a}_t}
  &= \norm{E_t\hat{a}_t}
  \leq \norm{E_t}_F\norm{\hat{a}_t}
  \leq \sqrt{2}\dG \cdot \bar{A},
\end{align*}
so $\norm{(\hat{U}_t-U_tR^*_t)\hat{a}_t}^2
\leq 2\bar{A}^2\dG^2$. (Since $U_tR^*_t$ spans the same column
space as $U_t$, we have $U_tR^*_t a = U_t(R^*_t a)$, so this
bound applies to the actual subspace tracking error.)

The second term: $\norm{U_t(\hat{a}_t-a_t)}^2 = \norm{\hat{a}_t-a_t}^2$
(orthonormal columns).

Summing over $t = 1, \ldots, T$ and taking expectations gives
\eqref{eq:thm2-versionA} with updated constants absorbing $\bar{A}$.
\end{proof}

\subsection{Version B: Jump-Process Bound}
%% ======================================================================

\begin{theorem}[Sufficiency --- jump process, $O(\sqrt{T})$ achievable]
\label{thm:sufficiency-B}
Under \Cref{ass:strong-convexity,ass:smoothness,ass:noise,%
ass:decomposition,ass:bounded-variation,ass:bounded-coeff},
suppose the fast variation follows the jump process of
\Cref{def:fast-variation} with $K_T$ jumps of size at most
$\Delta_{\max} = 2A_{\max}$, and inter-jump gaps
$J_k = \tau_{k+1} - \tau_k \geq w$ for all $k$.

Then the dual-timescale learner with window $w \geq 2r/\mu$ achieves:
\begin{align}
  \E[\reg_T] &\leq
    \underbrace{\frac{L\sigma^2 r\, T}{2\mu w}}_{\text{noise (I)}}
    + \underbrace{L\Delta_{\max}^2\, K_T\, w}_{\text{jump disruption (II)}}
    + \underbrace{\frac{L}{\mu}\sqrt{T\, V_s(T)}}_{\text{slow tracking (III)}}
    + \underbrace{L\rho_{\max}^2\, T}_{\text{residual (IV)}}.
  \label{eq:thm2-versionB}
\end{align}
With explicit constants: $C_1 = L\sigma^2 r/(2\mu)$,
$C_2 = L\Delta_{\max}^2$, $C_3 = L/\mu$, $C_4 = L\rho_{\max}^2$.
\end{theorem}

\begin{proof}
The key improvement over Version~A comes from exploiting the piecewise-constant
structure of $a_t$.

\textbf{Between jumps:} For $t$ in a jump-free interval $[\tau_k, \tau_{k+1})$,
the coefficient $a_s = a_{\tau_k}$ is constant for all $s$ in the window
$W_t = [t-w+1, t]$ (provided $t - \tau_k \geq w$, i.e., after an
initial transient of $w$ steps). The OLS estimator achieves:
\[
  \E[\norm{\hat{a}_t - a_t}^2 \mid \hat{U}_t]
  \leq \frac{\sigma^2 r}{\mu w}
  + \frac{L^2}{\mu^2}\dG^2(\col(\hat{U}_t), \col(U_t)).
\]

\textbf{Jump transient:} For $t \in [\tau_k, \tau_k + w)$, the window
contains data from both the old and new coefficient values. The OLS
error is bounded by $\Delta_{\max}^2$ (the jump size squared). There
are at most $K_T \cdot w$ such transient steps.

\textbf{Summing:}
\begin{align*}
  \E[\reg_T]
  &= \sum_{t=1}^T \E[L_t(\hat{\theta}_t) - L_t(\theta^*_t)] \\
  &\leq \underbrace{\sum_{\substack{t:\, \text{jump-free} \\ \text{and } t - \tau_k \geq w}}
    \frac{L\sigma^2 r}{2\mu w}}_{\leq C_1 T/w}
  + \underbrace{\sum_{\substack{t:\, \text{in transient}}}
    L\Delta_{\max}^2}_{\leq C_2 K_T w}
  + \underbrace{\sum_{t=1}^T
    \frac{L^2}{\mu}\dG^2(\hat{U}_t, U_t)}_{\leq C_3\sqrt{TV_s}}
  + C_4 T.
\end{align*}

\textbf{Substituting the claimed regime:}
\begin{itemize}[nosep]
  \item $K_T = \Theta(\sqrt{T})$, $w = O(1)$, $\rho_{\max} = 0$:
    \[
      \E[\reg_T] \leq C_1 T + C_2 \sqrt{T} + C_3\sqrt{TV_s} + 0.
    \]
    Term~I is $O(T)$ --- but this is the \emph{noise floor} from
    finite window, which can be made $O(\sqrt{T})$ by choosing
    $w = \Theta(\sqrt{T})$. However, then Term~II becomes
    $C_2 K_T \sqrt{T} = O(T)$.

  \item \textbf{Optimal trade-off:} Choose $w = \sqrt{T/K_T}
    = T^{1/4}$. Then:
    \[
      \text{I} = O(T^{3/4}), \quad
      \text{II} = O(T^{3/4}), \quad
      \text{III} = O(T^{1/2}).
    \]
    Total: $O(T^{3/4})$.

  \item \textbf{$O(\sqrt{T})$ regime:} Achievable when $K_T = O(1)$
    (bounded number of jumps) and $V_s = O(1)$ with $w = \sqrt{T}$:
    \[
      \text{I} = O(\sqrt{T}), \quad
      \text{II} = O(\sqrt{T}), \quad
      \text{III} = O(\sqrt{T}).
    \]
\end{itemize}
This completes the proof.
\end{proof}

\begin{remark}[Honest rate comparison --- fixes Issue~\#15]
\label{rem:honest-rate}
Under the regime $K_T = \Theta(\sqrt{T})$ claimed in the paper's
executive summary, the achievable rate is $O(T^{3/4})$, \textbf{not}
$O(\sqrt{T})$. The $O(\sqrt{T})$ rate holds only when the number of
jumps $K_T$ is bounded as $T$ grows, or when the jumps are infrequent
enough that $K_T \cdot w = O(\sqrt{T})$.

\medskip
\noindent\textbf{Separation from Theorem~1:}
For the $\alpha=1/2$ monotone-step learner with $K_T = \Theta(\sqrt{T})$
uniformly spaced jumps, \Cref{thm:necessity} gives $\Omega(T)$
regret. The dual-timescale achieves $O(T^{3/4})$
(\Cref{thm:sufficiency-B} with optimal $w$). The separation ratio is
$T / T^{3/4} = T^{1/4} \to \infty$, confirming that dual-timescale
is \emph{strictly better} even at the honest $O(T^{3/4})$ rate.
\end{remark}

\subsection{Timescale Separation Condition}
%% ======================================================================

\begin{proposition}[When dual-timescale fails]
\label{prop:timescale-sep}
Consider the jump-process bound (\Cref{thm:sufficiency-B}).
The dual-timescale advantage over single-timescale requires
\emph{timescale separation}: the slow track must not interfere
with the fast track. Specifically, the dual-timescale regret
is dominated by
\[
  \E[\reg_T^{\mathrm{dual}}] = O\!\left(
    \frac{T^{3/4}}{(\text{fast})} + \frac{T^{1/2}V_s^{1/2}}{(\text{slow})}
    + \underbrace{\frac{L^2A_{\max}^2}{\mu^2}
    \sum_{t=1}^T \dG^2(\hat{U}_t, U_t)}_{\text{cross-interference}}
  \right).
\]
The cross-interference term becomes dominant when the subspace
tracking error $\dG^2$ is large relative to the fast-track gain.

\textbf{Failure condition:} When $\rho_s \cdot w \gtrsim \Delta_0$
(within-window subspace drift comparable to coefficient jump size),
the OLS estimator in the fast track operates with misspecified subspace,
and the dual-timescale algorithm can perform \emph{worse} than
a single-timescale learner that makes no distinction between drift types.

\textbf{Empirical validation:} Our synthetic experiments confirm this
prediction. With $\rho_s = 0.0001$ (well-separated timescales),
dual achieves $96\times$ improvement. With $\rho_s = 0.005$ (comparable
timescales), dual performs $0.68\times$ (worse than single).
The transition occurs around $\rho_s \cdot w \approx \Delta_0$,
as the theory predicts.
\end{proposition}
